class Ship {
    constructor(name, hitPoints, damageReceived, color) {
        this.name = name;
        this.hitPoints = hitPoints;
        this.damageReceived = damageReceived;
        this.color = color;
        this.height = 50px;
        this.width = 70px;
        this.isDestroyed = false;
    }

    makeShip() {
        let ship = document.createElement('div')
        ship.style.width = this.width;
        ship.style.height = this.height;
        ship.style.textAlign = "center";
        ship.style.backgroundColor = this.color
        ship.innerHTML = `
            <h3>${this.name}</h3>
            <h5>${this.hitPoints}</h5>
        `
        document.body.appendChild(ship)
    }

    updateShip(ship) {
        ship.innerHTML = `
            <h3>${this.name}</h3>
            <h5>${this.hitPoints}</h5>
        `
        ship.style.backgroundColor = this.color
    }

    damageShip(ship) {
        this.hitPoints -= this.damageReceived;
        if (this.hitPoints <= 0) {
            destroyShip(ship)
        } else {
            this.updateShip()
        }
    }

    destroyShip(ship) {
        this.color = "black"
        this.hitPoints = "0"
        this.isDestroyed = true
        this.updateShip(ship)
    }
}
